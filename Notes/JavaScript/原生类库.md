//   随机数
//1. 简单的闭区间随机整数
parseInt(Math.random() * (b - a + 1)) + a;

//2. 返回一个范围内的随机数
function getRandomArbitrary(min, max) {
  return Math.random() * (max - min) + min;
}

//3. 返回一个介于min和max之间的整型随机数
function getRandomInt(min, max) {
    //Math.floor()向下取整
  return Math.floor(Math.random() * (max - min + 1) + min);
}
